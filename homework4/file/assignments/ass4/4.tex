\section{Question 4}

\begin{question}
  What are the active stations (status = online) with an average rating
  (extra.score) greater than or equal
  to 4?
\end{question}

\begin{answer}

  \begin{lstlisting}[language=js]
db.stations
    .find({
        "extra.status": "online", "extra.score": {
            "$gte": 4
        }
    }, {"_id": 0, "name": 1})
    .sort({"name": 1})
    .map(function (row) {
        return row.name;
    })
  \end{lstlisting}
